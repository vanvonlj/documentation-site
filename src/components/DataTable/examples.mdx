# DataTable Component Examples

## Basic Usage

```tsx
import DataTable from '@site/src/components/DataTable';

<DataTable
  data={[
    { name: 'John', age: 30, role: 'Developer' },
    { name: 'Jane', age: 25, role: 'Designer' },
    { name: 'Bob', age: 35, role: 'Manager' },
  ]}
  columns={[
    { key: 'name', header: 'Name' },
    { key: 'age', header: 'Age' },
    { key: 'role', header: 'Role' },
  ]}
/>
```

## TypeScript Type Safety

```tsx
import DataTable, { Column } from '@site/src/components/DataTable';

interface GearItem {
  item: string;
  slot: string;
  sockets: string;
  have: boolean;
}

const columns: Column<GearItem>[] = [
  { key: 'item', header: 'Item' },
  { key: 'slot', header: 'Gear Slot' },
  { key: 'sockets', header: 'Sockets' },
  {
    key: 'have',
    header: 'Have',
    render: (value: boolean, row: GearItem) => (
      <span>{value ? '✓' : '✗'} {row.item}</span>
    ),
  },
];

const gearData: GearItem[] = [
  { item: 'Helmet', slot: 'Head', sockets: 'x2', have: true },
];

<DataTable<GearItem> data={gearData} columns={columns} />
```

## Expandable Rows

```tsx
import DataTable from '@site/src/components/DataTable';

<DataTable
  data={[
    {
      item: "Loyalty's Mantle",
      slot: 'Helm',
      sockets: 'Neo-Ohm',
      have: true,
      details: {
        stats: '+500 Armor, +100 Intelligence',
        source: 'Drops from World Boss',
        notes: 'Best in slot for casters'
      }
    },
    {
      item: 'Temerity',
      slot: 'Pants',
      sockets: 'Emerald - Royal x2',
      have: false,
      details: {
        stats: '+300 Armor, Barrier Generation',
        source: 'Random world drop',
        notes: 'Essential for defensive builds'
      }
    },
  ]}
  columns={[
    { key: 'item', header: 'Item' },
    { key: 'slot', header: 'Gear Slot' },
    { key: 'sockets', header: 'Sockets' },
    {
      key: 'have',
      header: 'Have',
      render: (value) => (
        <span style={{ color: value ? '#49cc90' : '#f93e3e' }}>
          {value ? '✓' : '✗'}
        </span>
      ),
    },
  ]}
  expandable={true}
  renderExpandedRow={(row) => (
    <div style={{ padding: '0.5rem' }}>
      <h4 style={{ marginTop: 0 }}>{row.item} Details</h4>
      <p><strong>Stats:</strong> {row.details.stats}</p>
      <p><strong>Source:</strong> {row.details.source}</p>
      <p><strong>Notes:</strong> {row.details.notes}</p>
    </div>
  )}
/>
```

## Props API

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `data` | `T[]` | **required** | Array of data objects to display |
| `columns` | `Column<T>[]` | **required** | Column configuration array |
| `sortable` | `boolean` | `true` | Enable sorting functionality |
| `filterable` | `boolean` | `true` | Enable filtering functionality |
| `striped` | `boolean` | `true` | Alternate row colors |
| `hoverable` | `boolean` | `true` | Highlight rows on hover |
| `compact` | `boolean` | `false` | Reduce padding for smaller tables |
| `expandable` | `boolean` | `false` | Enable expandable rows with toggle button |
| `renderExpandedRow` | `(row: T) => ReactNode` | `undefined` | Function to render expanded row content |
| `headerless` | `boolean` | `false` | Hide the table header (thead) completely |
| `footerless` | `boolean` | `false` | Hide the table footer (row count) completely |

### Column Configuration

| Property | Type | Description |
|----------|------|-------------|
| `key` | `string` | Key to access value in data object |
| `header` | `string` | Column header text |
| `sortable` | `boolean` | Override global sortable setting for this column |
| `filterable` | `boolean` | Override global filterable setting for this column |
| `render` | `(value, row) => ReactNode` | Custom cell renderer function |
